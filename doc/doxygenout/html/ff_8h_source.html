129/* Filesystem object structure (FATFS) */
130
131typedef struct {
132 BYTE fs_type; /* Filesystem type (0:not mounted) */
133 BYTE pdrv; /* Associated physical drive */
134 BYTE n_fats; /* Number of FATs (1 or 2) */
135 BYTE wflag; /* win[] flag (b0:dirty) */
136 BYTE fsi_flag; /* FSINFO flags (b7:disabled, b0:dirty) */
137 WORD id; /* Volume mount ID */
138 WORD n_rootdir; /* Number of root directory entries (FAT12/16) */
139 WORD csize; /* Cluster size [sectors] */
140#if FF_MAX_SS != FF_MIN_SS
141 WORD ssize; /* Sector size (512, 1024, 2048 or 4096) */
142#endif
143#if FF_USE_LFN
144 WCHAR* lfnbuf; /* LFN working buffer */
145#endif
146#if FF_FS_EXFAT
147 BYTE* dirbuf; /* Directory entry block scratchpad buffer for exFAT */
148#endif
149#if FF_FS_REENTRANT
150 FF_SYNC_t sobj; /* Identifier of sync object */
151#endif
152#if !FF_FS_READONLY
153 DWORD last_clst; /* Last allocated cluster */
154 DWORD free_clst; /* Number of free clusters */
155#endif
156#if FF_FS_RPATH
157 DWORD cdir; /* Current directory start cluster (0:root) */
158#if FF_FS_EXFAT
159 DWORD cdc_scl; /* Containing directory start cluster (invalid when cdir is 0) */
160 DWORD cdc_size; /* b31-b8:Size of containing directory, b7-b0: Chain status */
161 DWORD cdc_ofs; /* Offset in the containing directory (invalid when cdir is 0) */
162#endif
163#endif
164 DWORD n_fatent; /* Number of FAT entries (number of clusters + 2) */
165 DWORD fsize; /* Size of an FAT [sectors] */
166 LBA_t volbase; /* Volume base sector */
167 LBA_t fatbase; /* FAT base sector */
168 LBA_t dirbase; /* Root directory base sector/cluster */
169 LBA_t database; /* Data base sector */
170#if FF_FS_EXFAT
171 LBA_t bitbase; /* Allocation bitmap base sector */
172#endif
173 LBA_t winsect; /* Current sector appearing in the win[] */
174 BYTE win[FF_MAX_SS]; /* Disk access window for Directory, FAT (and file data at tiny cfg) */
175} FATFS;

179/* Object ID and allocation information (FFOBJID) */
180
181typedef struct {
182 FATFS* fs; /* Pointer to the hosting volume of this object */
183 WORD id; /* Hosting volume mount ID */
184 BYTE attr; /* Object attribute */
185 BYTE stat; /* Object chain status (b1-0: =0:not contiguous, =2:contiguous, =3:fragmented in this session, b2:sub-directory stretched) */
186 DWORD sclust; /* Object data start cluster (0:no cluster or root directory) */
187 FSIZE_t objsize; /* Object size (valid when sclust != 0) */
188#if FF_FS_EXFAT
189 DWORD n_cont; /* Size of first fragment - 1 (valid when stat == 3) */
190 DWORD n_frag; /* Size of last fragment needs to be written to FAT (valid when not zero) */
191 DWORD c_scl; /* Containing directory start cluster (valid when sclust != 0) */
192 DWORD c_size; /* b31-b8:Size of containing directory, b7-b0: Chain status (valid when c_scl != 0) */
193 DWORD c_ofs; /* Offset in the containing directory (valid when file object and sclust != 0) */
194#endif
195#if FF_FS_LOCK
196 UINT lockid; /* File lock ID origin from 1 (index of file semaphore table Files[]) */
197#endif
198} FFOBJID;

203
204typedef struct {

206 BYTE flag; /* File status flags */
207 BYTE err; /* Abort flag (error code) */
208 FSIZE_t fptr; /* File read/write pointer (Zeroed on file open) */
209 DWORD clust; /* Current cluster of fpter (invalid when fptr is 0) */
210 LBA_t sect; /* Sector number appearing in buf[] (0:invalid) */
211#if !FF_FS_READONLY

214#endif
215#if FF_USE_FASTSEEK
216 DWORD* cltbl; /* Pointer to the cluster link map table (nulled on open, set by application) */

218#if !FF_FS_TINY

273/* File function return code (FRESULT) */
274
275typedef enum {
276 FR_OK = 0, /* (0) Succeeded */
277 FR_DISK_ERR, /* (1) A hard error occurred in the low level disk I/O layer */
278 FR_INT_ERR, /* (2) Assertion failed */
279 FR_NOT_READY, /* (3) The physical drive cannot work */
280 FR_NO_FILE, /* (4) Could not find the file */
281 FR_NO_PATH, /* (5) Could not find the path */
282 FR_INVALID_NAME, /* (6) The path name format is invalid */
283 FR_DENIED, /* (7) Access denied due to prohibited access or directory full */
284 FR_EXIST, /* (8) Access denied due to prohibited access */
285 FR_INVALID_OBJECT, /* (9) The file/directory object is invalid */
286 FR_WRITE_PROTECTED, /* (10) The physical drive is write protected */
287 FR_INVALID_DRIVE, /* (11) The logical drive number is invalid */
288 FR_NOT_ENABLED, /* (12) The volume has no work area */
289 FR_NO_FILESYSTEM, /* (13) There is no valid FAT volume */
290 FR_MKFS_ABORTED, /* (14) The f_mkfs() aborted due to any problem */
291 FR_TIMEOUT, /* (15) Could not get a grant to access the volume within defined period */
292 FR_LOCKED, /* (16) The operation is rejected according to the file sharing policy */
293 FR_NOT_ENOUGH_CORE, /* (17) LFN working buffer could not be allocated */
294 FR_TOO_MANY_OPEN_FILES, /* (18) Number of open files > FF_FS_LOCK */
295 FR_INVALID_PARAMETER /* (19) Given parameter is invalid */
296} FRESULT;
